\section{Intro}

  \begin{frame}
    \frametitle{Recap}
    \begin{alertblock}{CLEWs: What it is, what is has}
      \begin{itemize}
        \item Partial equilibrium
        \item Inputs: Prices, demands, basic resource, precip., temp.
        \item Technology: Production, cost function, inputs and outputs
        \item Outputs: Allocations, final costs, resource use, emissions, particulates
      \end{itemize}
    \end{alertblock}
    \begin{block}{OG-PHL: What it is, what it has}
      \begin{itemize}
        \item General equilibrium
        \item Inputs: household, firm, government parameters
        \item Technology: utility functions, production functions
        \item Outputs: prices, allocations, microeconomic and macroeconomic time series
      \end{itemize}
    \end{block}
  \end{frame}

\section{PEP}

  \begin{frame}
    \frametitle{}
    \begin{block}{Philippine Energy Plan 2023--2050 \href{https://legacy.doe.gov.ph/pep}{https://legacy.doe.gov.ph/pep}}
      \begin{itemize}
        \item ``...attain an inclusive and \textbf{equitable economic growth}... through the provision of secure, sustainable, and resilient energy strategies.''
        \item ``...energy sector's goal to ... attaining \textbf{clean energy} future.''
        \item 35\% and 50\% renewable energy share in power generation by 2030 and 2040
        \item 10\% EV penetration for road transport (motorcycles, cars, jeepneys) by 2040
        \item Resumption of indigenous oil and gas exploration
        \item Introduction of liquefied natural gas (LNG) portfolio
        \item Establishment of strategic petroleum reserves
        \item Exploration of Hydrogen's potential
      \end{itemize}
    \end{block}
  \end{frame}
